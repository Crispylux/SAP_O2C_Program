*&---------------------------------------------------------------------*
*&      Form  ZZ_SET_DISPLAY_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM zz_set_display_data .
*===============
* PO를 기준으로 뒤 문서를 조회해야 한다!!
*===============
*PO->SO->DO->GI->BI

  CHECK gt_itab IS NOT INITIAL.

  LOOP AT gt_itab ASSIGNING FIELD-SYMBOL(<gs_itab>).
    DATA: lv_wadat TYPE zlikp19-WADAT_IST.

    CLEAR: <gs_itab>-sosta,
           <gs_itab>-dosta,
           <gs_itab>-gista,
           <gs_itab>-bista,
           <gs_itab>-vaeln,
           <gs_itab>-vleln,
           <gs_itab>-vfeln.

    "SO 가지고 옴 (VAELN)
    SELECT SINGLE VBELN INTO <gs_itab>-VAELN
    FROM ZVBAK19
    WHERE BSTNK = <gs_itab>-EBELN.

    IF sy-subrc = 0.
      <gs_itab>-SOSTA = 'S'.           "초록불 바꾸기


      "DO 가지고 옴 (VLELN)
      SELECT SINGLE VBELN INTO <gs_itab>-VLELN
      FROM ZLIPS19
      WHERE VGBEL = <gs_itab>-VAELN.

      IF sy-subrc = 0.
        <gs_itab>-DOSTA = 'S'.         "초록불 바꾸기


        "GI 가지고 옴 (LIKP의 WADAT_IST)
        CLEAR lv_wadat.
        SELECT SINGLE wadat_ist INTO lv_wadat
        FROM ZLIKP19
        WHERE vbeln = <gs_itab>-vleln.

        IF sy-subrc = 0 AND lv_wadat IS NOT INITIAL.
          <gs_itab>-GISTA = 'S'.        "초록불 바꾸기


          "BI 가지고 옴 (VFELN)
          SELECT SINGLE VBELN INTO <gs_itab>-VFELN
          FROM ZVBRP19
          WHERE VGBEL = <gs_itab>-VLELN.

          IF sy-subrc = 0.
            <gs_itab>-BISTA = 'S'.       "초록불 바꾸기
          ENDIF.
        ENDIF.
      ENDIF.
    ENDIF.

  ENDLOOP.
ENDFORM.                    " ZZ_SET_DISPLAY_DATA


*&---------------------------------------------------------------------*
*&      Form  zz_check_atpc_auth
*&---------------------------------------------------------------------*
*       Auto Process (ATPC) 권한
*----------------------------------------------------------------------*
FORM zz_check_atpc_auth.
  DATA: lv_loekz TYPE ZT19ACOM10-LOEKZ,
        lv_v3  TYPE zt19acom10-VALU3,
        lv_v4  TYPE zt19acom10-VALU4.

  CLEAR: gv_no_atpc, lv_loekz, lv_v3, lv_v4.

  "Delete Flag
  SELECT SINGLE loekz valu3 valu4 INTO (lv_loekz, lv_v3, lv_v4)
  FROM ZT19ACOM10
  WHERE PCODE = 'SD_CREATE_AUTO' AND
        UNAME = sy-uname.

  "아무것도 안담김
  IF sy-subrc <> 0.
    gv_no_atpc = 'X'.
    RETURN.
  ENDIF.

  "DELETE FLAG 체크면 무조건 안됨
  IF lv_loekz = 'X'.
    gv_no_atpc = 'X'.
    RETURN.
  ENDIF.

  "V3,V4 다 체크되어야 ATPC 허용
  IF lv_v3 = 'X' AND lv_v4 = 'X'.
    CLEAR gv_no_atpc. "사용O
  ELSE.
    gv_no_atpc = 'X'. "사용X
  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*&      Form  ZZ_CREATE_SO
*&---------------------------------------------------------------------*
*       Create SO
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM zz_create_so .
  DATA: lt_rows   TYPE TABLE OF lvc_s_roid, "몇 번째 행을 선택했는지 알려줌
        ls_po_h   TYPE zekko19,
        lt_po_i   TYPE TABLE OF zekpo19,
        ls_return TYPE bapiret2.

*===============
* 행 선택 안되어있을 경우
*===============
  CALL METHOD go_av2000_1->get_selected_rows IMPORTING et_row_no = lt_rows[].
  IF lt_rows IS INITIAL.
    MESSAGE s000(oo) WITH '행을 선택하세요' DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

*===============
* 행 번호로 데이터 가져오기
*===============
  READ TABLE lt_rows INDEX 1 ASSIGNING FIELD-SYMBOL(<ls_rows>).
  READ TABLE gt_itab INDEX <ls_rows>-row_id ASSIGNING FIELD-SYMBOL(<gs_itab>).
  CHECK sy-subrc = 0.

*===============
* 중복검사
*===============
  IF <gs_itab>-vaeln IS NOT INITIAL.
    IF gv_atpc_check = 0. "아님
      MESSAGE i000(oo) WITH 'SO already exists.'.
      RETURN. "서브루틴 종료
    ELSE. "ATPC이면
      EXIT. "현재 실행 중인 블록만 빠져나감
    ENDIF.
  ENDIF.


*===============
* 헤더, 아이템 걸러내기
*===============
  CLEAR ls_po_h.
  MOVE-CORRESPONDING <gs_itab> TO ls_po_h.

  SELECT * INTO CORRESPONDING FIELDS OF TABLE lt_po_i
    FROM EKPO
    WHERE ebeln = <gs_itab>-ebeln.

  IF lt_po_i IS INITIAL.
    MESSAGE e000(oo) WITH '아이템이 없습니다.'.
    RETURN.
  ENDIF.


*===============
* SO 만들기! (method: ZZ_GET_SO_RTN)
*===============
  CALL METHOD zcl19_lec_auto_plan=>zz_get_so_rtn
    EXPORTING
      is_po_h   = ls_po_h     " ZEKK019 구조
      it_po_i   = lt_po_i      " ZEKPO19 구조
    IMPORTING
      es_return = ls_return.





  IF ls_return-type = 'S'.              "성공이면
    <gs_itab>-vaeln = ls_return-field.
    <gs_itab>-sosta = 'S'.
    MESSAGE s000(oo) WITH 'SO 생성됨!: ' ls_return-field.
  ELSE.
    MESSAGE s000(oo) WITH 'SO 생성 실패...' DISPLAY LIKE 'E'.
  ENDIF.

ENDFORM.                    " ZZ_CREATE_SO

*----------------------------------------------------------------------*
* 핫스팟 표시해둔 열(디스플레이 되게!)
*----------------------------------------------------------------------*
FORM AV2000_1_CELL_CLICK
  USING pv_av_name
        pc_gubun   " 'B' 버튼 'D' 떠블클릭 'H' 핫스팟
        ps_row    LIKE lvc_s_row
        ps_col    LIKE lvc_s_col
        ps_row_no LIKE lvc_s_roid.

  READ TABLE GT_ITAB ASSIGNING FIELD-SYMBOL(<gs_itab>)
                         INDEX ps_row_no-row_id.            "현재 클릭한 행 골라줌
  CHECK sy-subrc EQ 0.

  CASE ps_col-fieldname.
    WHEN 'EBELN'. "SHOW PO
      SET PARAMETER ID 'BES' FIELD <gs_itab>-ebeln.
      CALL TRANSACTION 'ME23N' AND SKIP FIRST SCREEN.


    WHEN 'VAELN'. "DISPLAY SO
      EXPORT gv_display_60 = 'X' TO MEMORY ID 'DISPLAY_SO'. "이름, 버튼 없애기(20)

      IF <gs_itab>-VAELN IS NOT INITIAL.
        SUBMIT ZR19A00020
        WITH so_vbeln = <gs_itab>-VAELN
        WITH so_erdat IN se_aedat
        AND RETURN.
      ENDIF.


    WHEN 'VLELN'. "DISPLAY DO
      EXPORT gv_display_60 = 'X' TO MEMORY ID 'DISPLAY_DO'. "이름, 버튼 없애기(40)

      IF <gs_itab>-VLELN IS NOT INITIAL.
        SUBMIT ZR19A00040
        WITH so_vbeln = <gs_itab>-VLELN
        WITH so_erdat IN se_aedat
        AND RETURN.
      ENDIF.

    WHEN 'VFELN'. "DISPLAY BI
      IF <gs_itab>-VFELN IS NOT INITIAL.
      "SUBMIT으로 50의 스크린을 띄운다(선택한 줄 vfeln값만 선택해서 띄움)
        SUBMIT zr19a00050
        WITH so_vbeln = <gs_itab>-VFELN
        WITH so_erdat IN se_aedat
        AND RETURN.
      ENDIF.
  ENDCASE.

ENDFORM.
